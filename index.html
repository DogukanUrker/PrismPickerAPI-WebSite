{
          "rgb": [
              0,
              0,
              0
          ]

/convert/hex/rgba/{hexCode}/{alpha}
{
          "rgba": [
              0,
              0,
              0,
              1.0
          ]
}

/convert/rgb/hex/{r}/{g}/{b}
{
          "hex": "#000000"

Mixer

Two RGB

/mixer/rgb/{r1}/{g1}/{b1}/{r2}/{g2}/{b2}
{
          "rgb": [
              127,
              127,
              127
          ]
}

Two RGBA

/mixer/rgba/{r1}/{g1}/{b1}/{a1}/{r2}/{g2}/{b2}/{a2}
{
          "rgba": [
              127,
              127,
              127,
              0.625
          ]
}

Two Hex

/mixer/hex/{hex1}/{hex2}
{
          "hex": "#7f7f7f"
}

Multiple Hex [GET]

/mixer/hex?color={color1}&color={color2}&color={color3}...
{
          "hex": "#666666"
}
